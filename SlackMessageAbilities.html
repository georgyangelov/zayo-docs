def react_with(reaction, message: nil)
  reaction = reaction.to_s.gsub(/\:/, '')
  channel, _ = response_defaults

  if !message && !context.last_message_ts
    raise 'Cannot react before a message or on a different channel'
  end

  message_ts = message ? message.ts : context.last_message_ts

  begin
    @app.client.web_client.reactions_add(
      name: reaction,
      channel: channel,
      timestamp: message_ts,
      as_user: true
    )
    logger.info "Reacted with ':#{reaction.to_s}:' to '#{message_ts}' in channel '#{channel}'"
  rescue Slack::Web::Api::Errors::SlackError => error
    if error.message == 'already_reacted'
      logger.info "Already reacted with ':#{reaction.to_s}:' to '#{message_ts}' in channel '#{channel}'"
    else
      raise
    end
  end
end

def say(text, vars: nil, channel: nil, thread: nil)
  channel, thread = response_defaults channel, thread

  text = any_string text.to_s.split('.') if text.is_a? Symbol
  text = text % vars if vars

  response = @app.client.web_client.chat_postMessage(
    text: text,
    channel: channel,
    thread_ts: thread,
    as_user: true
  )
  logger.info "Sent message '#{text}' to '#{channel}'#{thread && ":#{thread}"}"

  Message.new(response.channel, response.ts)
end
