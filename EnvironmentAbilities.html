def find_channel_id(channel_name)
  channel_name = channel_name.sub(/^\#/, '')
  return $1 if channel_name =~ /<#(.*?)\|.*>/

  channel = slack_channels.find { |k, c| c.name == channel_name || c.id == channel_name }

  if channel
    channel.last.id
  else
    # TODO: Wat do?
    channel_name
  end
end

def member_of_channel?(name)
  channel_id = find_channel_id(name)
  channel = slack_channels[channel_id]

  return nil unless channel

  channel.is_member
end
